376int BFS(Grafo grafo, int inicial) {
377 if (grafo.numVertices < 1) return 0;
378 if (!grafo.vertices) return 0;
379 if (!grafo.vertices->prox) return 0;
380
381 int *fila = malloc(grafo.numVertices * sizeof(int));
382 if (!fila) return 0;
383 int front = 0, rear = 0;
384
385 fila[rear++] = inicial;
386
387 while (front < rear) {
388 int codigo = fila[front++];
389 Vertice* atual = encontrarVerticePorCod(grafo, codigo);
390 if (atual && !atual->visitado) {
391 atual->visitado = 1;
392 printf("%c|%d: (X: %d, Y : %d)", atual->antena.frequencia, atual->codigo, atual->antena.x, atual->antena.y);
393 for (Adjacente* adj = atual->adjacentes; adj; adj = adj->prox) {
394 Vertice* vertAdj = encontrarVerticePorCod(grafo, adj->codigo);
395 if (vertAdj && !vertAdj->visitado) {
396 int estaNaFila = 0;
397 for (int k = front; k < rear; k++) {
398 if (fila[k] == adj->codigo) {
399 estaNaFila = 1;
400 break;
401 }
402 }
403 if (!estaNaFila) fila[rear++] = adj->codigo;
404 }
405 }
406 if (front < rear) {
407 printf(" -> ");
408 }
409 }
410 }
411 free(fila);
412 return 1;

466int listarTodosCaminhos(Grafo grafo, int inicial, int final) {
