234 if (linha[x] != '.' && linha[x] != '#')
235 adicionarVertice(&grafo, x, y, linha[x]);
236 }
237 }

240 for (Vertice* proximo = atual->prox; proximo; proximo = proximo->prox) {
241 if (atual->antena.frequencia == proximo->antena.frequencia) {
242 adicionarAdjacente(&grafo, atual->codigo, proximo->codigo);
243 adicionarAdjacente(&grafo, proximo->codigo, atual->codigo);
244 }
245 }
246 }
247
248 fclose(file);

356int BFS(Grafo grafo, int inicial) {
357 if (grafo.numVertices < 1) return 0;
358 if (!grafo.vertices) return 0;
359 if (!grafo.vertices->prox) return 0;
360
361 int *fila = malloc(grafo.numVertices * sizeof(int));
362 if (!fila) return 0;
363 int front = 0, rear = 0;
364
365 fila[rear++] = inicial;
366
367 while (front < rear) {
368 int codigo = fila[front++];
369 Vertice* atual = encontrarVerticePorCod(grafo, codigo);
370 if (atual && !atual->visitado) {
371 atual->visitado = 1;
372 printf("%c|%d: (X: %d, Y : %d)", atual->antena.frequencia, atual->codigo, atual->antena.x, atual->antena.y);
373 for (Adjacente* adj = atual->adjacentes; adj; adj = adj->prox) {
374 Vertice* vertAdj = encontrarVerticePorCod(grafo, adj->codigo);
375 if (vertAdj && !vertAdj->visitado) {
376 int estaNaFila = 0;
377 for (int k = front; k < rear; k++) {
378 if (fila[k] == adj->codigo) {
379 estaNaFila = 1;
380 break;
381 }
382 }
383 if (!estaNaFila) fila[rear++] = adj->codigo;
384 }
385 }
386 if (front < rear) {
387 printf(" -> ");
388 }
389 }
390 }
391 free(fila);
392 return 1;
393}
